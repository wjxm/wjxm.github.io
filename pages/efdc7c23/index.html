django序列化组件

什么是序列化

序列化：将python数据类型转换成json

反序列化：将json转换成python

Django-JsonResponse序列化

作用

通过JsonResponse 将数据json序列化，并发送

为什么要用JsonResponse而不是json模块

JsonResponse可以序列化更多的django 相关数据

例如：通过json.dumps 无法序列化date格式，所以通过jsonresponse实现

参数

  • jsonresponse传入列表，要加上safe=false参数
  • 设置编码json_dumps_params={"ensure_ascii": False}
点击查看

实现简单的封装数据，然后通过jsonresponse发送

views.py

from django.shortcuts import render
from django.views import View
from django.http import HttpResponse, JsonResponse
from django.core import serializers
from .models import Book, Publisher
import json
class BookView(View):
    def get(self, request):
        book_list = Book.objects.values("id", "title", "category", "pub_time", "publisher")
        book_list = list(book_list)
        ret = []
        # 组装成一个列表，内部包含字典的这么个数据结构
        for book in book_list:
            publisher_id = book["publisher"]
            publisher_obj = Publisher.objects.filter(id=publisher_id).first()
            
            book["publisher"] = {
                "id": publisher_id,
                "title": publisher_obj.title
            }
            ret.append(book)
        # ret = json.dumps(book_list, ensure_ascii=False)
        return JsonResponse(ret, safe=False, json_dumps_params={"ensure_ascii": False})

django serializers序列化

序列化出来的值,包含有原始的外键值,未经处理，不适合前端使用

views.py

from django.core import serializers
class BookView(View):
    def get(self, request):
        book_list = Book.objects.all()
        ret = serializers.serialize("json", book_list, ensure_ascii=False)
        return HttpResponse(ret)

djangorestframework实现序列化

  1. 首先安装 pip install djangorestframework

  2. 在setting中注册rest_framework app

  3. 声明一个序列化器对象,创建一个文件serializers.py,在里面写入一个序列化类(像model这样的序列化类),继承serializers.Serializer

  4. 在views.py 中类继承APIView类，

    1. get方法，接受浏览器get请求，用序列化器对象来序列化数据,然后通过response 发送，
    2. post方法，接收浏览器post请求，将接收到的数据通过序列化器来校验，校验成功.is_valid()结果为真，结果在 request.data 里
    3. put方法，

  5. 接收到浏览器返回的序列化数据，也通过序列化器来校验，

  6. 对外键关系序列化时,需要在序列化器里将外键关

  7. 当序列化和反序列化的类型不一样时,用read_only 和write_only 分别生成两个字段.

序列化的验证

  • 字段本身的，类型，长度等校验
  • 局部钩子
  • 全局钩子
  • 校验函数，放入字段参数类使用

# 类似于model语法
from rest_framework import serializers
from .models import Book

class PublisherSerializer(serializers.Serializer):
    # 序列化器的校验规则
    id = serializers.IntegerField()

class AuthorSerializer(serializers.Serializer):
    id = serializers.IntegerField()
    name = serializers.CharField(max_length=32)

            "w_category": 1,

            "publisher_id": 1,
            "author_list": [1, 2]
        }


    data = {
        "title": "Alex的使用教程2"

def my_validate(value):
    # 校验器，在字段中添加，权重比钩子函数高
    if "敏感信息" in value.lower():
        raise serializers.ValidationError("不能含有敏感信息")
    else:
        return value


class BookSerializer(serializers.Serializer):
    id = serializers.IntegerField(required=False) 不校验
    # title 接上面的验证函数
    title = serializers.CharField(max_length=32, validators=[my_validate])
    CHOICES = ((1, "Python"), (2, "Go"), (3, "Linux"))
    category = serializers.ChoiceField(choices=CHOICES, source="get_category_display", read_only=True)
    w_category = serializers.ChoiceField(choices=CHOICES, write_only=True)
category 在前端显示的是python， 如果从前端传数据回来给，categotry只知道1，2，3，不知道python，这个时候就新建一个w_category字段来接收，write_only=True 反序列化字段

    pub_time = serializers.DateField()

    publisher = PublisherSerializer(read_only=True)
    publisher_id = serializers.IntegerField(write_only=True)
    author = AuthorSerializer(many=True, read_only=True) many=true 多对多
    author_list = serializers.ListField(write_only=True)

    def create(self, validated_data):
        # 通过save调用
        book = Book.objects.create(title=validated_data["title"], category=validated_data["w_category"],
                                pub_time=validated_data["pub_time"], publisher_id=validated_data["publisher_id"])
        book.author.add(*validated_data["author_list"])
        return book

    def update(self, instance, validated_data):
        # 序列化器参数有partial=True,save()时调用此方法
        # instance 要更新的对象
        # validated_data 更新的数据
        instance.title = validated_data.get("title", instance.title)
        instance.category = validated_data.get("category", instance.category)
        instance.pub_time = validated_data.get("pub_time", instance.pub_time)
        instance.publisher_id = validated_data.get("publisher_id", instance.publisher_id)
        if validated_data.get("author_list"):
            # author_list这个字段需要先验证下是否存在
            instance.author.set(validated_data["author_list"])
        instance.save()
        return instance

    def validate_title(self, value):
        # 钩子函数，validate_开头加字段，单独对title进行校验
        if "python" not in value.lower():
            raise serializers.ValidationError("标题必须含有python")
        return value

    def validate(self, attrs):
        # 全局钩子函数，多字段联合校验
        if attrs["w_category"] == 1 and attrs["publisher_id"] == 1:
            return attrs
        else:
            raise serializers.ValidationError("分类以及标题不符合要求")

ModelSerializer序列化，通过继承serializers.ModelSerializer来实现

优点： 不用写create方法，已经默认写了create

可以引用model字段，在model字段不满足的情况下，可以自己写字段

外键的获取：

  • 通过depth = 1 # 外键关系中，只向下找一层，缺点是会获取所有字段
  • 通过SerializerMethodField返回一个自己组装的字段，可以自定义要获取的字段，减少冗余

优点： 不用重写updat,create 等方法

class BookSerializer(serializers.ModelSerializer):
    # SerializerMethodField 方法字段，调用get_+字段名  方法
    category_display = serializers.SerializerMethodField(read_only=True)
    publisher_info = serializers.SerializerMethodField(read_only=True)
    authors = serializers.SerializerMethodField(read_only=True)

    def get_category_display(self, obj):
        # orm的方法，获取category的中文 ，字段要添加参数 read_only=True
        return obj.get_category_display()

    def get_authors(self, obj):
        authors_query_set = obj.author.all()
        return [{"id": author_obj.id, "name": author_obj.name} for author_obj in authors_query_set]

    def get_publisher_info(self, obj):
        # obj 是我们序列化的每个Book对象
        publisher_obj = obj.publisher
        return {"id": publisher_obj.id, "title": publisher_obj.title}

    class Meta:
        model = Book
        # fields = ["id", "title", "pub_time"]
        fields = "__all__"
        # depth = 1  # 外键关系中，只向下找一层
        
        # 额外的参数
        extra_kwargs = {"category": {"write_only": True}, "publisher": {"write_only": True},
                        "author": {"write_only": True}}

from rest_framework.views import APIView
from rest_framework.viewsets import GenericViewSet
from rest_framework.response import Response
from .serializers import BookSerializer
class BookView(APIView):
    """
    拿django写的时候是继承view,现在继承apiview
    返回函数也变成了上面导入的response
    """
    def get(self, request):
        # 传入一个
        # book_obj = Book.objects.first()
        # ret = BookSerializer(book_obj)
        # 传入多个  many=True
        book_list = Book.objects.all()
        ret = BookSerializer(book_list, many=True)
        return Response(ret.data)


    def post(self, request):
        print(request.data)
        #通过序列化器进行反序列化
        serializer = BookSerializer(data=request.data)
        if serializer.is_valid():
            serializer.save()  # 执行序列化器里面自己定义的create方法
            return Response(serializer.data)
        else:
            return Response(serializer.errors)


class BookEditView(APIView):

    def get(self, request, id):
        # 查看单条数据，根据传入的id,进行查找返回
        book_obj = Book.objects.filter(id=id).first()
        ret = BookSerializer(book_obj)
        return Response(ret.data)
      
    def put(self, request, id):
        book_obj = Book.objects.filter(id=id).first()
        # 通过BookSerializer序列化器进行校验
    	# book_obj要更新的对象，
        # data=request.data 要更新的数据，
        # partial=True  允许部分更新，对象的save()方法调用update方法
        serializer = BookSerializer(book_obj, data=request.data, partial=True)
        if serializer.is_valid():
            serializer.save()
            return Response(serializer.data)
        else:
            return Response(serializer.errors)

    def delete(self, request, id):
        book_obj = Book.objects.filter(id=id).first()
        book_obj.delete()
        return Response("")
