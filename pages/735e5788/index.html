冒泡排序

  • 列表每两个相邻的数，如果前面比后面大，则交换这两个数。
  • 一趟排序完成后，则无序区减少一个数，有序区增加一个数。
  • 代码关键点：趟、无序区范围

# _*_coding:utf-8_*_
import random
def bubble_sort(li):
    for i in range(len(li)-1):  #第i趟
        exchange = False
        for j in range(len(li)-i-1):
            if li[j] > li[j+1]:
            	# 升序排序与降序排序取决于 > ,< 好 ,>是升序
                li[j], li[j+1] = li[j+1], li[j]
                # python独有的交换语法,内部是元组的交换 
                exchange = True
        if not exchange:
        	# 如果一趟结束后,没有发生交换,说明后面的数据都是有序的,不用再计算了
            return

def select_sort(li):
    for i in range(len(li)-1): # i是第几趟
        min_loc = i
        for j in range(i+1, len(li)):
            # 找出最小的数,从最左边的无序区开始找,大循环一次,有序区加一,无序区减一

                # 标记出最小的值的下标

插入排序

初始时手里(有序区)只有一张牌

每次(从无序区)摸一张牌,插入到手里已有牌的正确位置 (左边,或右边)

# _*_coding:utf-8_*_
def insert_sort(li):
    # 将摸到的牌与手里的最右边的牌对比,如果比手里的牌大,
    for i in range(1, len(li)): #i 表示摸到的牌的下标
        tmp = li[i]
        j = i - 1 #j指的是手里的牌的下标
        while j >= 0 and li[j] > tmp:
            # 手里最大的牌比摸到的牌大
            li[j+1] = li[j]
            # 手里最大的牌就往右移,给摸到的小牌腾位置
            j -= 1
            # 将指针调到手里第二(依次减)大牌的位置,在下一轮,while循环时,比较摸到的牌与第二大的牌
            
            # 如果摸到的牌比手里最大的牌大,直接跳出循环,并把摸到的牌放到当前牌的右边位置
        li[j+1] = tmp
        print(li)

def partition(li, left, right):
    
   """
   找一个tmp 把li[left]存起来, 从最右边开始往左,查找小于tmp的值,比如这个值是li[left-3],放到li[left],接下来从li[left +1 ]的位置,从左往右找大于tem的值,放到 li[left-3]的位置,然后再找右边,以此类推,直到left > right
   """
    tmp = li[left]
    while left < right:
        while left < right and li[right] >= tmp: #从右面找比tmp小的数,找到退出循环,没找到继续往左边走
            right -= 1      # 往左走一步
        li[left] = li[right] #把右边的值写到左边空位上
        # print(li, 'right')
        while left < right and li[left] <= tmp:
            left += 1
        li[right] = li[left] #把左边的值写到右边空位上
        # print(li, 'left')
    li[left] = tmp      # 把tmp归位
    return left

def _quick_sort(li, left, right):
    if left<right:  # 至少两个元素
        # left ,right ,表示左下标,和右下标,两个下标的中间区域就是 当前需要排序的区域
        mid = partition(li, left, right)
        _quick_sort(li, left, mid-1)
        _quick_sort(li, mid+1, right)

def quick_sort(li):
    _quick_sort(li, 0, len(li)-1)

li = list(range(10000, 0, -1))

quick_sort(li)

堆排序前传 树与二叉树

树的概念

  • 树是一种数据结构比如：目录结构
  • 树是一种可以递归定义的数据结构
  • 树是由n个节点组成的集合:
    • 如果n=0，那这是一棵空树;
    • 如果n>0，那存在1个节点作为树的根节点，其他节点可以分为m个集合，每个集合本身又是一棵树。

一些概念

根节点、叶子节点

树的深度（高度）

树的度 ,某个分叉最多的节点,它的分叉数就是树的度

孩子节点/父节点

子树 ,大树的一个树杈

二叉树的概念

满二叉树：

一个二叉树，如果每一个层的结点数都达到最大值，则这个二叉树就是满二叉树。

完全二叉树：

叶节点只能出现在最下层和次下层，并且最下面一层的结点都集中在该层最左边的若

二叉树的存储方式

二叉树的存储方式（表示方式）

  • 顺序存储方式

堆

堆：一种特殊的完全二叉树结构

大根堆：一棵完全二叉树，满足任一节点都比其孩子节点大

小根堆：一棵完全二叉树，满足任一节点都比其孩子节点小

# _*_coding:utf-8_*_

def sift(li, low, high):
    """
    :param li: 列表
    :param low: 堆的根节点位置
    :param high: 堆的最后一个元素的位置
    :return: 
    
    此函数实现堆的向下调整过程,首先将父节点赋值给中间变量, 根据父节点 获取两个子节点, 如果有右孩子节点,先比较两个子节点中选出最大的那个,然后最大的那个子节点再与父节点比,如果父节点值大于子节点,将中间变量赋值给父节点,反之将子节点的值赋值给父节点,然后将这个子节点作为父节点,进行下一轮的比较
    """
    i = low            # i最开始指向根节点
    j = 2 * i + 1      # j开始是左孩子
    tmp = li[low]      # 把堆顶存起来
    while j <= high:   # 只要j位置有数
        if j + 1 <= high and li[j+1] > li[j]: # 如果右孩子有并且比较大
            j = j + 1        # j指向右孩子
        if li[j] > tmp:
            li[i] = li[j]    # 如果孩子,比父亲大,将孩子放到父亲位置
            i = j            # 往下看一层,把这个孩子节点看做父节点,再向孙子比较
            j = 2 * i + 1    # j指向当前父节点的孩子节点
        else:                # tmp更大，把tmp放到i的位置上
            li[i] = tmp      # 把tmp放到某一级领导位置上
            break
    else:
        li[i] = tmp          # 把tmp放到叶子节点上


def heap_sort(li):
    n = len(li)
    for i in range((n-2)//2, -1, -1):
        # 子节点 i    父节点  (i-1)//2
        # 子节点 n-1  父节点  (n-1-1)//2
        # i表示建堆的时候最末尾的一个根节点, 
        sift(li, i, n-1)
    # 建堆完成了
    
    # 挨个出数
    for i in range(n-1, -1, -1):
        # i 指向当前堆的最后一个元素
        li[0], li[i] = li[i], li[0]
        # 根节点下来,最后的元素上去,把右边当做有序区,左边当做无序区
        sift(li, 0, i - 1) #i-1是新的high

堆排序应用-topk问题

现在有n个数，设计算法得到前k大的数。(k<n) 解决思路:

  • 排序后切片
  • 排序(冒泡,选择,插入)
  • 堆排序思路

topk实现思路

  • 取列表前k个元素建立一个小根堆。堆顶就是目前第k大的数。
  • 依次向后遍历原列表，对于列表中的元素，如果小于堆顶，则忽略该元素；如果大于堆顶，则将堆顶更换为该元素，并且对堆进行一次调整;遍历列表所有元素后，倒序弾出堆顶。

代码实现

# _*_coding:utf-8_*_
# 比较排序
def sift(li, low, high):
    # 通道改变 <= 符号来获得大根堆,小根堆
    i = low
    j = 2 * i + 1
    tmp = li[low]
    while j <= high:
        if j + 1 <= high and li[j+1] < li[j]:
            j = j + 1
        if li[j] < tmp:
            li[i] = li[j]
            i = j
            j = 2 * i + 1
        else:
            break
        li[i] = tmp


def topk(li, k):
    heap = li[0:k]
    for i in range((k-2)//2, -1, -1):
        # 先从li中获取k个数组成第一个堆
        sift(heap, i, k-1)
    # 1.建堆
    for i in range(k, len(li)-1):
        # 再遍历去掉k个元素的列表, 如果遍历到的元素比小根堆的最小数要大,就将这个数放入小更堆
        if li[i] > heap[0]:
            heap[0] = li[i]
            sift(heap, 0, k-1)

    # 2.遍历
    for i in range(k-1, -1, -1):
        # 将小根堆,解析出从大到小的列表
        heap[0], heap[i] = heap[i], heap[0]
        sift(heap, 0, i - 1)
    # 3.出数
    return heap

def merge(li, low, mid, high):
    i = low
    j = mid + 1
    ltmp = []
    while i<=mid and j<=high: # 只要左右两边都有数
        if li[i] < li[j]:
            ltmp.append(li[i])
            i += 1
        else:
            ltmp.append(li[j])
            j += 1
    # while执行完，肯定有一部分没数了
    while i <= mid:
        ltmp.append(li[i])
        i += 1
    while j <= high:
        ltmp.append(li[j])
        j += 1
    li[low:high+1] = ltmp

希尔排序

希尔排序(Shell Sort)是一种分组插入排序算法。
首先取一个整数d1=n/2,将元素分为d个组，每组相邻量元素之间距离为d，在各组内进行直接插入排序；取第二个整数d2=d/2,重复上述分组排序过程，直到d;=1,即所有元素在同一组内进行直接插入排序。
希尔排序每趟并不使某些元素有序，而是使整体数据越来越接近有序；最后一趟排序使得所有数据有序。

# _*_coding:utf-8_*_

def insert_sort_gap(li, gap):
    for i in range(gap, len(li)): #i 表示摸到的牌的下标
        tmp = li[i]
        j = i - gap #j指的是手里的牌的下标
        while j >= 0 and li[j] > tmp:
            li[j+gap] = li[j]
            j -= gap
        li[j+gap] = tmp

def shell_sort(li):
    d = len(li) // 2
    while d >= 1:
        insert_sort_gap(li, d)
        d //= 2

def count_sort(li, max_count=100):
    count = [0 for _ in range(max_count+1)]
    for val in li:
        count[val] += 1
    li.clear()
    for ind, val in enumerate(count):

桶排序

#! /usr/bin/env python
# -*- coding: utf-8 -*-
# Date: 2017/12/10

import random


def bucket_sort(li, n=100, max_num=10000):
    buckets = [[] for _ in range(n)] # 创建桶
    for var in li:
        i = min(var // (max_num // n), n-1) # i 表示var放到几号桶里
        buckets[i].append(var) # 把var加到桶里边
        # 保持桶内的顺序
        for j in range(len(buckets[i])-1, 0, -1):
            # 通过冒泡排序,在放入数据的时候就进行冒泡排序
            if buckets[i][j] < buckets[i][j-1]:
                buckets[i][j], buckets[i][j-1] = buckets[i][j-1], buckets[i][j]
            else:
                break
    sorted_li = []
    for buc in buckets:
        sorted_li.extend(buc)   # 把每个桶拼接起来
    return sorted_li


li = [random.randint(0,10000) for i in range(100000)]
# print(li)
li = bucket_sort(li)
print(li)

def partition(li, left, right):
    tmp = li[left]
    while left < right:
        while left < right and li[right] >= tmp: #从右面找比tmp小的数
            right -= 1      # 往左走一步
        li[left] = li[right] #把右边的值写到左边空位上
        # print(li, 'right')
        while left < right and li[left] <= tmp:
            left += 1
        li[right] = li[left] #把左边的值写到右边空位上
        # print(li, 'left')
    li[left] = tmp      # 把tmp归位
    return left


def _quick_sort(li, left, right):
    if left<right:  # 至少两个元素
        mid = partition(li, left, right)
        _quick_sort(li, left, mid-1)
        _quick_sort(li, mid+1, right)

@cal_time
def quick_sort(li):
    _quick_sort(li, 0, len(li)-1)

@cal_time
def radix_sort(li):
    max_num = max(li) # 最大值 9->1, 99->2, 888->3, 10000->5
    it = 0
    while 10 ** it <= max_num:
        buckets = [[] for _ in range(10)]
        for var in li:
            # 987 it=1  987//10->98 98%10->8;    it=2  987//100->9 9%10=9
            digit = (var // 10 ** it) % 10
            buckets[digit].append(var)
        # 分桶完成
        li.clear()
        for buc in buckets:
            li.extend(buc)
        # 把数重新写回li
        it += 1
